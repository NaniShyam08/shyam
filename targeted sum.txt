#include <stdio.h>

int main() {
    int arr[] = {2, 4, 5, 7, 1, 3, 6, 8};
    int n = sizeof(arr) / sizeof(arr[0]);
    int target, found = 0;
    
    // Get target sum from user
    printf("Enter the target sum: ");
    scanf("%d", &target);
    
    // Find pairs that sum to target
    for (int i = 0; i < n; i++) {
        for (int j = i + 1; j < n; j++) {
            if (arr[i] + arr[j] == target) {
                printf("Elements: %d, %d\n", arr[i], arr[j]);
                found = 1;
            }
        }
    }
    
    // Print -1 if no pairs found
    if (found == 0) {
        printf("-1\n");
    }
    
    return 0;
}
